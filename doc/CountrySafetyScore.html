def self.for_safety_level(level)
  case level
  when "Very Safe"
    very_safe
  when "Generally Safe"
    generally_safe
  when "Partly Safe"
    partly_safe
  when "Not Safe"
    not_safe
  else
    all
  end
end

def badge_color
  case gpi_score
  when 0...1.6
    "success"  # Green
  when 1.6...2.15
    "info"     # Blue
  when 2.15...2.7
    "warning"  # Yellow
  else
    "danger"   # Red
  end
end

def safety_description
  case safety_level

  end
end
